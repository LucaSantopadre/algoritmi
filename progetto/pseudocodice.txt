algoritmo sampleMedianSelect(list, left, right, m) --->  int chiave

    V = m elementi random da list[left:right]     O(m)

    lista = [(CHIAVE,valore),(CHIAVE,valore), ... ]   memorizza tutte le coppie CHIAVE/VALORE di V in m tuple,          O(m)

    mid = medianDictInsertionSort(lista, m)    calcola il mediano di V e ritorna la CHIAVE      O(m^2)

    return mid

algoritmo medianDictInsertionSort(dict, dim) ---> int chiave

    insertionSort su VALORE         O(m^2)

    calcola mediano di dict su chiave  mediano = (chiave = dim/2)      O(1)

    return CHIAVE del mediano
